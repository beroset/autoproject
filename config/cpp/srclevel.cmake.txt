cmake_minimum_required(VERSION 3.20)
{extras}
if (CMAKE_CXX_COMPILER_ID MATCHES "MSVC")
    # warning level 4 and all warnings as errors
    target_compile_options({projname} PRIVATE /W4 /WX)
elsef(CMAKE_CXX_COMPILER_ID MATCHES "GNU|Clang")
    # lots of warnings and all warnings as errors
    target_compile_options({projname} PRIVATE -Wall -Wextra -pedantic -Werror)
endif()
add_executable({projname} {srcnames})
target_compile_features({projname} PUBLIC cxx_std_17)
target_link_libraries({projname} {libraries})
